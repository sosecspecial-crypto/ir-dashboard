function generateMockData() {
  const data = {};

// Fetch data from Google Apps Script backend
async function fetchFromBackend() {
  if (!CONFIG.BACKEND_URL) {

    console.error('Backend error:', err);
    document.getElementById('nav-sync').textContent = 'Offline — using cache';
    if (!Object.keys(state.data).length) state.data = generateMockData();
  }
}

// Upload photo record to backend
async function uploadToBackend(zone, head, dateStr, files) {
  const key = `${zone}_${head}`;
  if (!state.data[key]) state.data[key] = { dates: [], total: 0 };
  if (!state.data[key].dates.includes(dateStr)) {
    state.data[key].dates.push(dateStr);
    state.data[key].total = state.data[key].dates.length;
  }

  if (!CONFIG.BACKEND_URL) {
    console.log('Mock upload:', zone, head, dateStr, files?.length, 'files');
    return { success: true };
  }

  const formData = new FormData();
  formData.append('action', 'upload');
  formData.append('zone', zone);
  formData.append('head', head);
  formData.append('date', dateStr);
  if (files) Array.from(files).forEach(f => formData.append('files', f));

  try {
    const res = await fetch(CONFIG.BACKEND_URL, { method: 'POST', body: formData });
    return await res.json();
  } catch (err) {
    console.error('Upload error:', err);
    return { success: false, error: err.message };
  }
}

function getUploadDates(zone, head) {
  return state.data[`${zone}_${head}`]?.dates || [];
}

function getZoneTotal(zone) {
  return HEADS.reduce((sum, h) => sum + (state.data[`${zone}_${h.id}`]?.dates.length || 0), 0);
}

function isPast(mi, d) {
  const t = CONFIG.TODAY;
  if (mi < t.getMonth()) return true;
  if (mi === t.getMonth() && d <= t.getDate()) return true;
  return false;
}

function zoneStatus(zone) {
  const t = getZoneTotal(zone);
  if (t === 0) return 'ns';
  if (t <= 5) return 'pe';
  if (t <= 20) return 'ac';
  return 'ex';
}

// RENDER CALENDAR FOR ONE HEAD

function renderCalendar(zone, head) {
  const uploadedDates = new Set(getUploadDates(zone, head));
  let greenDays = 0, redDays = 0;

  const monthCards = MONTHS.map((mn, mi) => {
    const dayHeaders = ['Su','Mo','Tu','We','Th','Fr','Sa'].map(d => `<div class="dh">${d}</div>`).join('');
    const blanks = Array(FD[mi]).fill(`<div class="dy blank"></div>`).join('');
    let mGreen = 0, mTotal = 0;
    const days = Array.from({length: MDAYS[mi]}, (_, i) => {
      const d = i + 1;
      const dateStr = `2025-${String(mi+1).padStart(2,'0')}-${String(d).padStart(2,'0')}`;
      const past = isPast(mi, d);
      const has = uploadedDates.has(dateStr);
      if (past) { mTotal++; if (has) { mGreen++; greenDays++; } else { redDays++; } }
      let cls = 'dy ';
      if (!past) cls += 'future';
      else if (has) cls += 'uploaded';
      else cls += 'missed';
      return `<div class="${cls}" title="${mn} ${d} — ${!past?'future':has?'uploaded':'no upload'}">${d}</div>`;
    }).join('');
    const pct = mTotal > 0 ? Math.round(mGreen / mTotal * 100) : 0;
    return `<div class="mcard ${mGreen>0?'has-uploads':''}">
      <div class="mhead"><span>${mn.slice(0,3)}</span><span style="font-weight:400;font-size:10px;color:#aaa">${mGreen}/${mTotal}</span></div>
      <div class="mbar"><div class="mbar-fill" style="width:${pct}%"></div></div>
      <div class="daygrid">${dayHeaders}${blanks}${days}</div>
    </div>`;
  }).join('');

  return { html: monthCards, greenDays, redDays };
}

// ─────────────────────────────────────────
// UPLOAD HANDLERS
// ─────────────────────────────────────────
let pendingFiles = {};

function handleDrop(e, zone, head) {
  e.preventDefault();
  document.getElementById(`dropzone-${zone}-${head}`).classList.remove('drag');
  pendingFiles[`${zone}_${head}`] = e.dataTransfer.files;
}

function handleUpload(files, zone, head) {
  pendingFiles[`${zone}_${head}`] = files;
}

async function submitUpload(zone, head) {
  const monthEl = document.getElementById(`month-${zone}-${head}`);
  const dayEl   = document.getElementById(`day-${zone}-${head}`);
  const fbEl    = document.getElementById(`fb-${zone}-${head}`);
  const msgEl   = document.getElementById(`fbmsg-${zone}-${head}`);
  const month   = monthEl?.value;
  const day     = String(parseInt(dayEl?.value || 1)).padStart(2, '0');

  if (!month) { alert('Please select a month'); return; }

  const dateStr = `2025-${month}-${day}`;
  const files   = pendingFiles[`${zone}_${head}`];
  const result  = await uploadToBackend(zone, head, dateStr, files);

  if (result.success !== false) {
    msgEl.textContent = `${files?.length || 1} file(s) saved to ${zone}/${head}/${dateStr} — synced`;
    fbEl.classList.add('show');
    setTimeout(() => fbEl.classList.remove('show'), 5000);
    pendingFiles[`${zone}_${head}`] = null;
    // Re-render to update calendar
    renderZonePanel(zone);
    renderStats();
  }
}

function openGPhotos(zone, head) {
  window.open(`${CONFIG.GPHOTOS_BASE}${encodeURIComponent('IR ' + zone + ' ' + head)}`, '_blank');
}

// ─────────────────────────────────────────
// NAVIGATION
// ─────────────────────────────────────────
function setZone(za) {
  state.activeZone = za;
  state.activeHead = 'all';
  renderZoneTabs();
  renderStats();
  if (za === 'all') renderAllZones();
  else renderZonePanel(za);
  if (za !== 'all') {
    const t = document.getElementById('zone-tabbar').querySelector('.ztab.active');
    if (t) t.scrollIntoView({ inline: 'center', behavior: 'smooth' });
  }
}

function setHead(hid) {
  state.activeHead = hid;
  renderZonePanel(state.activeZone);
}

  document.getElementById('nav-date').textContent =
    CONFIG.TODAY.toLocaleDateString('en-IN', { month: 'long', year: 'numeric' });

    renderStats();
    if (state.activeZone === 'all') renderAllZones();
    else renderZonePanel(state.activeZone);
  }, CONFIG.REFRESH_INTERVAL);
}
